Token Helper Module

Functions for generating, storing, and validating one-time tokens.

Constants

const TOKEN_EXPIRATION_MS = 24 * 60 * 60 * 1000

Tokens expire after 24 hours (86400000 ms).

Functions

generateToken()

function generateToken() {
    return crypto.randomBytes(32).toString('hex')
}

createToken()

async function createToken() {
    const tokenString = generateToken()
    const token = new Token({ token: tokenString })
    await token.save()
    return tokenString
}

tokenError(errorMessage)

function tokenError(errorMessage) {
    return {
        success: false,
        error: errorMessage,
        name: null,
        email: null,
        message: null
    }
}

validateToken(providedToken)

async function validateToken(providedToken) {
    const actualToken = await Token.findOne({ token: providedToken })
    if (!actualToken) return tokenError("This link could not be found.")
    if (actualToken.used) return tokenError("This link has already been used.")
    const ageInMs = Date.now() - actualToken.createdAt
    if (ageInMs > TOKEN_EXPIRATION_MS) return tokenError("This link has expired.")
    actualToken.used = true
    await actualToken.save()
    return { success: true, token: actualToken }
}

Usage Example

const { createToken, validateToken } = require('../helpers/tokenHelpers')

const token = await createToken()
const result = await validateToken(token)
if(result.success) console.log('Token valid', result.token)
else console.log('Token error', result.error)
